# Configurações do Servidor
PORT=3000
HOST=0.0.0.0
NODE_ENV=production

# MongoDB - Configure com a URL do seu banco de dados
# Exemplo local: mongodb://localhost:27017/app_beneficios
# Exemplo remoto: mongodb://usuario:senha@host:porta/database
MONGO_URI=mongodb://localhost:27017/app_beneficios

# JWT Secret - Gere uma string aleatória segura
# Você pode gerar com: node -e "console.log(require('crypto').randomBytes(64).toString('hex'))"
JWT_SECRET=seu_jwt_secret_super_seguro_aqui

# CORS - URLs permitidas (separadas por vírgula)
# Para produção, coloque APENAS os domínios autorizados a chamar a API
# Exemplo produção: https://seu-app.com,https://api.seu-app.com
# Para desenvolvimento local: http://localhost:3000,http://192.168.1.15:3000
# Evite usar '*' em produção.
CORS_ORIGIN=http://localhost:3000
